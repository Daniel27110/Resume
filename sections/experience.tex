\cvitem{\href{https://hdl.handle.net/1992/77102}{Détection d'Anomalies Cliniques en Unités de Soins Intensifs Pédiatriques}}{Jan 2025 -- Juil 2025}{Projet \dotsep Hôpital Militaire Central de Bogotá}{Bogotá, Colombie}
\begin{itemize}[itemsep=1pt, topsep=1pt, leftmargin=15pt]
  \item Développement d'un système de surveillance prédictive temps réel pour l'analyse des signes vitaux au chevet de patients pédiatriques, améliorant le rappel de 12\% et réduisant le taux de faux positifs de 38,7\%.

\end{itemize}
